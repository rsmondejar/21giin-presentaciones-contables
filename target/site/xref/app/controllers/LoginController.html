8   import app.dao.UserDao;
9   import app.entities.User;
10  import helpers.Dialog;

13  /**
14   * Login Controller.
15   *
16   * @author raulsm
17   */
18  public class LoginController extends BaseController {
19  
20      public static void loggin(String name, String password) {
21  
22          Boolean isValid = false;
23  
24          // Check if user exists in database
25          User user = null;
26          try {
27              user = UserDao.findByLoginAndPassword(name, password);
28              isValid = true;
29          } catch (Exception exception) {
30              Log.error(exception);
31          }
32  
33          if (!isValid) {
34              Dialog.error("Usuario y/o contrase√±a incorrectos", "Error login");
35              return;
36          }
37  
38          // Store session user      
39          Main.setAuthUser(user);
40  
41          // Hide login view
42          Main.loginPanelHide();
43  
44          // Show Main Menu
45          MainController mainController = new MainController();
46          mainController.menuPanelShow();
47      }
48  }
