5   package helpers;
6   
7   import javax.swing.JOptionPane;
8   
9   /**
10   * Modal Helper.
11   *
12   * @author raulsm
13   */
14  public class Dialog {
15  
16      public static void info(String message, String title) {
17          custom(message, title, JOptionPane.INFORMATION_MESSAGE);
18      }
19  
20      public static void error(String message, String title) {
21          custom(message, title, JOptionPane.ERROR_MESSAGE);
22      }
23      
24      public static void warning(String message, String title) {
25          custom(message, title, JOptionPane.WARNING_MESSAGE);
26      }
27  
28      /**
29       * Brings up a dialog that displays a message using a default icon
30       * determined by the <code>messageType</code> parameter.
31       *
32       * @param message the <code>Object</code> to display
33       * @param title the title string for the dialog
34       * @param messageType the type of message to be displayed:
35       * <code>ERROR_MESSAGE</code>, <code>INFORMATION_MESSAGE</code>,
36       * <code>WARNING_MESSAGE</code>, <code>QUESTION_MESSAGE</code>, or
37       * <code>PLAIN_MESSAGE</code>
38       */
39      public static void custom(String message, String title, int messageType) {
40          JOptionPane.showMessageDialog(
41                  null,
42                  message,
43                  title,
44                  messageType
45          );
46      }
47  }
