55      /**
56       * Load Municipalities Documents Uploaded.
57       */
58      private void loadMunicipalitiesDocumentUploaded() {
59          DefaultTableModel model = (DefaultTableModel) jTableMunicipalitiesDocuments.getModel();
60  
61          List<MunicipalityTotalDocument> items = InformationController.municipalitiesPresentationsTotalDocuments();
62          
63          // Add data
64          Object[] rowData = new Object[model.getColumnCount()];
65  
66          if (items != null && !items.isEmpty()) {
67              for (int i = 0; i < items.size(); i++) {
68                  rowData[0] = items.get(i).getMunicipalityName();
69                  rowData[1] = items.get(i).getTotalDocuments();
70  
71                  model.addRow(rowData);
72              }
73          }
74  
75      }

77      /**
78       * Load Data Convocatories To Table
79       *
80       * @param convocatories Convocatories
81       * @param table jTable
82       */
83      private void loadDataConvocatoriesToTable(List<Convocatory> convocatories, JTable table) {
84          DefaultTableModel model = (DefaultTableModel) table.getModel();
85  
86          // Add data
87          Object[] rowData = new Object[model.getColumnCount()];
88  
89          if (convocatories != null && !convocatories.isEmpty()) {
90              for (int i = 0; i < convocatories.size(); i++) {
91                  rowData[0] = convocatories.get(i).getId();
92                  rowData[1] = convocatories.get(i).getName();
93                  rowData[2] = convocatories.get(i).getTotalPresentations();
94  
95                  model.addRow(rowData);
96              }
97          }
98      }
