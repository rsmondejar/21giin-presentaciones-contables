117     /**
118      * Set Documents Types Required For Convocatory.
119      *
120      * @param convocatory Convocatory
121      */
122     private void setDocumentsTypesRequiredForConvocatory(Convocatory convocatory) {
123         List<DocumentType> documentTypes = convocatory.getDocumentsTypes();
124 
125         if (documentTypes != null) {
126             for (DocumentType documentType : documentTypes) {
127                 int documentTypeID = documentType.getId();
128                 if (documentTypeID == 1) {
129                     jButtonDocumentType1.setEnabled(true);
130                 }
131                 if (documentTypeID == 2) {
132                     jButtonDocumentType2.setEnabled(true);
133                 }
134                 if (documentTypeID == 3) {
135                     jButtonDocumentType3.setEnabled(true);
136                 }
137                 if (documentTypeID == 4) {
138                     jButtonDocumentType4.setEnabled(true);
139                 }
140                 if (documentTypeID == 5) {
141                     jButtonDocumentType5.setEnabled(true);
142                 }
143             }
144         }
145     }

147     /**
148      * Get Documents Uploaded
149      *
150      * @param convocatory
151      * @return
152      */
153     private List<PresentationDocumentType> getDocumentsUploaded(Convocatory convocatory) {
154         List<DocumentType> documentTypes = convocatory.getDocumentsTypes();
155         List<PresentationDocumentType> presentationDocumentTypes = new ArrayList<>();
156 
157         if (documentTypes != null) {
158             for (DocumentType documentType : documentTypes) {
159                 int documentTypeID = documentType.getId();
160 
161                 PresentationDocumentType presentationDocumentType = new PresentationDocumentType();
162                 presentationDocumentType.setDocumentTypeId(documentTypeID);
163                 byte[] documentContent = null;
164                 String fileName = null;
165                 String fileExtension = null;
166 
167                 if (documentTypeID == 1 && documentType1.length > 0) {
168                     documentContent = documentType1;
169                     fileName = documentTypeFileName1;
170                     fileExtension = documentTypeFileExtension1;
171                 }
172                 if (documentTypeID == 2 && documentType2.length > 0) {
173                     documentContent = documentType2;
174                     fileName = documentTypeFileName2;
175                     fileExtension = documentTypeFileExtension2;
176                 }
177                 if (documentTypeID == 3 && documentType3.length > 0) {
178                     documentContent = documentType3;
179                     fileName = documentTypeFileName3;
180                     fileExtension = documentTypeFileExtension3;
181                 }
182                 if (documentTypeID == 4 && documentType4.length > 0) {
183                     documentContent = documentType4;
184                     fileName = documentTypeFileName4;
185                     fileExtension = documentTypeFileExtension4;
186                 }
187                 if (documentTypeID == 5 && documentType5.length > 0) {
188                     documentContent = documentType5;
189                     fileName = documentTypeFileName5;
190                     fileExtension = documentTypeFileExtension5;
191                 }
192 
193                 if (documentContent != null) {
194                     presentationDocumentType.setContent(documentContent);
195                     presentationDocumentType.setFileName(fileName);
196                     presentationDocumentType.setFileExtension(fileExtension);
197                     presentationDocumentTypes.add(presentationDocumentType);
198                 }
199             }
200         }
201 
202         return presentationDocumentTypes;
203     }

920     /**
921      * Download File
922      *
923      * @param documentType
924      * @throws Exception
925      */
926     private void downloadFile(int documentType) {
927         JFileChooser fileChooser = new JFileChooser();
928         byte[] documentContent;
929         String fileName = null;
930 
931         try {
932             switch (documentType) {
933                 case 1:
934                     documentContent = documentType1;
935                     fileName = documentTypeFileName1;
936                     break;
937                 case 2:
938                     documentContent = documentType2;
939                     fileName = documentTypeFileName2;
940                     break;
941                 case 3:
942                     documentContent = documentType3;
943                     fileName = documentTypeFileName3;
944                     break;
945                 case 4:
946                     documentContent = documentType4;
947                     fileName = documentTypeFileName4;
948                     break;
949                 case 5:
950                     documentContent = documentType5;
951                     fileName = documentTypeFileName5;
952                     break;
953                 default:
954                     throw new Exception("Document type option ['%d'] is not valid".formatted(documentType));
955             }
956 
957             fileChooser.setSelectedFile(new File(fileName));
958             int tampak = fileChooser.showSaveDialog(null);
959 
960             if (tampak == JFileChooser.APPROVE_OPTION) {
961                 File file = fileChooser.getSelectedFile();
962                 Blob blob = new SerialBlob(documentContent);
963                 InputStream stream = blob.getBinaryStream();
964 
965                 try ( OutputStream output = new FileOutputStream(file)) {
966                     byte[] buffer = new byte[4096];
967 
968                     while (stream.read(buffer) > 0) {
969                         output.write(buffer);
970                     }
971                 }
972             }
973         } catch (Exception ex) {
974             Dialog.error(ex.getMessage(), "Opci√≥n no valida");
975             Log.error(ex);
976         }
977     }

982      * @param fileNameLabel JLable File Name To Display de file path
983      * @param documentType Document Type
984      */
985     private void loadFileContent(JLabel fileNameLabel, int documentType) {
986         JFileChooser fileChooser = new JFileChooser();
987         fileChooser.setFileFilter(FileTypeFilterPDF.getFilter());
988         fileChooser.showOpenDialog(null);
989         File file = fileChooser.getSelectedFile();
990         String fileName = file.getAbsolutePath();
991 
992         try ( FileInputStream fileInputStream = new FileInputStream(file)) {
993             ByteArrayOutputStream bos = new ByteArrayOutputStream();
994             byte[] buffer = new byte[1024];
995 
996             for (int readNum; (readNum = fileInputStream.read(buffer)) != -1;) {
997                 bos.write(buffer, 0, readNum);
998             }
999 
1000             switch (documentType) {
1001                 case 1:
1002                     documentType1 = bos.toByteArray();
1003                     documentTypeFileName1 = file.getName();
1004                     documentTypeFileExtension1 = FilenameUtils.getExtension(fileName);
1005                     jButtonDocumentTypeDelete1.setEnabled(true);
1006                     jButtonDocumentTypeDownload1.setEnabled(true);
1007                     break;
1008                 case 2:
1009                     documentType2 = bos.toByteArray();
1010                     documentTypeFileName2 = file.getName();

1012                     jButtonDocumentTypeDelete2.setEnabled(true);
1013                     jButtonDocumentTypeDownload2.setEnabled(true);
1014                     break;
1015                 case 3:

1019                     jButtonDocumentTypeDelete3.setEnabled(true);
1020                     jButtonDocumentTypeDownload3.setEnabled(true);
1021                     break;
1022                 case 4:

1024                     documentTypeFileName4 = file.getName();
1025                     documentTypeFileExtension4 = FilenameUtils.getExtension(fileName);
1026                     jButtonDocumentTypeDelete4.setEnabled(true);
1027                     jButtonDocumentTypeDownload4.setEnabled(true);
1028                     break;
