180     @SuppressWarnings("unused")
181     private void jButtonUserEditActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButtonUserEditActionPerformed
182         try {
183 
184             int id = Integer.parseInt(jTextFieldUserId.getText());
185             User user = UserController.findById(id);
186             
187             UserEditJInternalFrame userEdit = new UserEditJInternalFrame();
188         
189             jDesktopPanelContainer.add(userEdit);
190         
191             // Load user info in view
192             userEdit.loadData(user);
193             
194             this.hide();
195             
196             userEdit.setVisible(true);
197         } catch (NumberFormatException e) {
198             Log.error(e);
199             Dialog.error(e.getMessage(), "Error editar municipio");
200         }
201     }//GEN-LAST:event_jButtonUserEditActionPerformed

203     @SuppressWarnings("unused")
204     private void jButtonUserDeleteActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButtonUserDeleteActionPerformed
205         int input = JOptionPane.showConfirmDialog(null, "Deseas eliminar?");
206         // 0=yes, 1=no, 2=cancel
207         
208         if (input == 0) {
209             // Delete
210             int id = Integer.parseInt(jTextFieldUserId.getText());
211             boolean status = UserController.delete(id);
212             
213             if (status) {
214                 Dialog.info("Usuario eliminado correctamente", "Info");
215                 
216                 this.hide();
217             } else {
218                 Dialog.error("Usuario no se ha eliminado", "Error");
219             }
220         }
221     }//GEN-LAST:event_jButtonUserDeleteActionPerformed
