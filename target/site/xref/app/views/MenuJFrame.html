37      /**
38       * Show or Hide Sections by User
39       * @param user Auth User
40       */
41      private void showHideSections(User user) {
42          int roleId = user.getRoleId();
43  
44          if (roleId != Role.ADMININISTRATOR) {
45              jMenuUsers.setVisible(false);
46              jMenuMunicipios.setVisible(false);
47          }
48  
49          switch (roleId) {
50              case Role.CUENTADANTE: {
51                  jMenuItemConvocatoriesCreate.setVisible(false);
52              }
53              break;
54              case Role.FISCAL: {
55                  jMenuMunicipios.setVisible(true);
56                  jMenuItemMunicipiosCreate.setVisible(false);
57                  jMenuItemConvocatoriesCreate.setVisible(false);
58              }
59              break;
60              case Role.FISCAL_GENERAL: {
61                  jMenuMunicipios.setVisible(true);
62                  jMenuItemMunicipiosCreate.setVisible(false);
63              }
64              break;
65          }
66      }
