122     private void jButtonMunicipioUpdateActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButtonMunicipioUpdateActionPerformed
123         try {
124 
125             int id = Integer.parseInt(jTextFieldMunicipioId.getText());
126             Municipality municipality = MunicipalityController.findById(id);
127 
128             municipality.setName(getInputName());
129             municipality.setCategoryId(Integer.parseInt(getInputCategory()));
130 
131             MunicipalityController.update(id, municipality);
132             
133             Dialog.info("Municipio actualizado correctamente", "Info");
134         } catch (NumberFormatException e) {
135             Log.error(e);
136             Dialog.error(e.getMessage(), "Error actualizando municipio");
137         }
138     }//GEN-LAST:event_jButtonMunicipioUpdateActionPerformed

140     /**
141      * Validate Inputs.
142      *
143      * @return True if inputs are valid
144      */
145     private Boolean validateInputs() {
146         return validateInputName() && validateInputCategory();
147     }
148 
149     /**
150      * Validate Input Name.
151      *
152      * @return
153      */
154     private Boolean validateInputName() {
155         String name = getInputName();
156         Boolean isValid = name.length() > 0 && name.length() <= 30;
157 
158         if (isValid || name.length() == 0) {
159             jTextFieldMunicipioName.setBackground(Color.white);
160         } else {
161             jTextFieldMunicipioName.setBackground(Color.orange);
162         }
163 
164         return isValid;
165     }
166 
167     /**
168      * Validate Input Category.
